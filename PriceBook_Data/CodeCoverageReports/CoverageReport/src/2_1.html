using System.Collections;

namespace PriceBook_Data
{
    public class Categories : IEnumerable<Category>
    {

        private readonly List<Category> _cats = new List<Category>();


        public Categories(bool getAll = false)
        {
            if (getAll)
            {
                using (ApplicationContext context = new ApplicationContext())
                {
                    //_cats = context.Category.ToDictionary();
                    foreach (var category in context.Category)
                    {
                        _cats.Add(category);
                    }
                }

                foreach (var cat in _cats)
                {
                    cat.IsAttached = true;
                }
            }
        }

        public int Count => _cats.Count;

        public Category this[int i] => _cats[i];

        public IEnumerator<Category> GetEnumerator()
        {
            return _cats.GetEnumerator();
        }

        [System.Diagnostics.CodeAnalysis.ExcludeFromCodeCoverageAttribute]
        IEnumerator IEnumerable.GetEnumerator()
        {
            return GetEnumerator();
        }

        public List<Category> ToList()
        {
            return _cats.ToList();
        }

        public Category[] ToArray()
        {
            return _cats.ToArray();
        }


    }
}
